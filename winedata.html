def knn_filled(testdata) :
    df = testdata
    #
    if df.price.isnull().any():#null
        data = pd.DataFrame()
        for i in range(0,df.shape[0],6):# 6010, 6
            data1 = df["price"][i:i+6].reset_index(drop = True).rename(index = str(i))
            data = pd.concat((data, data1), axis = 1) #dataframe
        data_incomplete = np.array(data)#array
        #
        filled_knn = KNN(k=3).fit_transform(data_incomplete)#knn
        data_complete = pd.DataFrame(filled_knn)#
        for i in range(0,data_complete.shape[1]):
            x.extend(data_complete[i].values.tolist())
    return x
